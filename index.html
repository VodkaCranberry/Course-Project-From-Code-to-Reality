07 // 中期创客项目：复古光学摩斯密码机 V5.0

项目简介：

基于 ESP32 主控，整合了光敏侦测与机械按键双路输入的摩斯密码解析系统，并在本地建立 Web Server，实现了实体敲击与网页端数据显示的交互互通。

团队成员与分工

本项目由 黄新雄 与 邓童 协作完成。具体分工涵盖了硬件电路搭建、3D外壳建模、C++ 解析逻辑编写及前端网页控制台的整合。

具体操作过程

  1. 硬件电路搭建与验证： 首先在面包板上搭建原型电路。利用光敏电阻与固定电阻串联构成 分压电路，接入 ESP32 的模拟输入引脚以采集光照阈值；同时接入机械按键并配置下拉电阻读取高低电平。
  2. 结构设计与组装： 根据面包板与主控板的实际尺寸，设计并使用 3D 打印机打印了专用的外壳。将接线完毕的面包板整体嵌入壳体，确保外部光线能准确照射到光敏电阻，按键留有操作孔位。
  3. 软件开发与算法调试：
    • 时间差算法： 放弃了容易引起死机的 delay() 方案，改用 millis() 函数记录触发的下降沿和上升沿，计算时间差。
    • 双模逻辑判定： 按键模式下，按下 50ms~2s 判定为“点(.)”，大于 2s 判定为“划(-)”；光感模式下，为了防抖，遮光 2~4s 才判定为“点(.)”，大于 4s 判定为“划(-)”。
    • 通信层： 使用 <WebServer.h> 搭建网页端，通过 /data 接口返回包含当前输入缓冲区和已解密字母的 JSON 数据，实现网页端每 500ms 的自动刷新反馈。

/**
 * 项目：复古光学摩斯密码机 V5.0
 * 核心逻辑：基于 millis() 的双模时间差计算
 */

#include <WiFi.h>
#include <WebServer.h>

const int LDR_PIN = 4;       
const int BUTTON_PIN = 6;    
const int LED_PIN = 15;      

const char *ssid = "TitanicRising";
const char *password = "20060426";

// --- 判定参数 (毫秒) ---
const int LIGHT_THRESHOLD = 7500;  
const unsigned long BTN_DASH_TIME = 2000;
const unsigned long LDR_DOT_MIN = 2000;  
const unsigned long LDR_DASH_MIN = 4000; 

WebServer server(80);
String morseBuffer = "";    
String decodedMessage = ""; 
unsigned long startTime = 0;
bool isActive = false;      
bool buttonMode = false;    

char decodeMorse(String s) {
  if (s == ".-") return 'A';   if (s == "-...") return 'B'; 
  if (s == "-.-.") return 'C'; if (s == "-..") return 'D';  
  if (s == ".") return 'E';    if (s == "..-.") return 'F';
  // ... 摩斯密码字典映射
  return '?';
}

void loop() {
  server.handleClient();
  
  // 状态轮询：根据当前模式读取物理输入
  bool triggered = buttonMode ? (digitalRead(BUTTON_PIN) == LOW) : (analogRead(LDR_PIN) >= LIGHT_THRESHOLD);

  if (triggered && !isActive) {
    startTime = millis(); // 记录起始时间
    isActive = true;
    digitalWrite(LED_PIN, HIGH);
  }

  if (!triggered && isActive) {
    unsigned long dur = millis() - startTime; // 计算持续时长
    isActive = false;
    digitalWrite(LED_PIN, LOW);

    // 逻辑判定树
    if (buttonMode) {
      if (dur >= BTN_DASH_TIME) morseBuffer += "-";
      else if (dur > 50) morseBuffer += "."; // 50ms 软件消抖
    } else {
      if (dur >= LDR_DASH_MIN) morseBuffer += "-"; 
      else if (dur >= LDR_DOT_MIN) morseBuffer += "."; 
    }
    Serial.println("当前缓冲区: " + morseBuffer);
  }
}

09 // 设计思维 (Design Thinking)

在硬件项目的开发中，设计思维指导了我们的完整流程：

  • Empathize (共情)： 从实际需求出发，明确需要解决什么问题。
  • Define (定义)： 锁定技术难点（如按键抖动、光线干扰）。
  • Ideate (构思)： 提出解决方案，比如在代码中加入基于时间的阈值过滤算法。
  • Prototype (原型)： 使用面包板或 3D 打印机快速制作粗略版本验证可行性。
  • Test (测试)： 软硬联调，不断根据输出结果修正代码逻辑。

项目构思与系统方案

期末项目的设计目标是制作一个具有实际应用意义的智能硬件作品。我们选择“植物养护”作为场景， 因为宿舍、实验室或办公室中的植物经常因为忘记浇水而枯萎。系统通过土壤湿度传感器采集模拟量， ESP32-S2 Mini 将原始 AD 值映射为 0%~100% 的湿度百分比，并根据设定阈值判断是否缺水。

为了提升交互性，系统同时提供两种反馈方式：一是 TFT 小屏幕本地显示时间、湿度、IP 地址和状态； 二是通过 WiFi 建立 Web Server，在同一局域网下使用浏览器访问控制台，实现远程查看和控制。 当湿度低于报警线时，蜂鸣器会间歇鸣叫，网页端也会显示红色警告，并允许手动静音。

硬件组成

  • 主控板： ESP32-S2 Mini，用于 WiFi 通信、传感器读取、网页服务和屏幕驱动。
  • 土壤湿度传感器： 通过模拟输入读取土壤含水量，并根据干湿标定值转换为湿度百分比。
  • 0.8 寸 TFT 彩屏： 使用 Adafruit_ST7735 库驱动，显示时间、湿度、状态与 IP 地址。
  • 蜂鸣器： 当湿度低于报警阈值时发出间歇报警声。
  • 杜邦线与面包板： 用于快速搭建原型电路和调试引脚连接。
  • 3D 打印外壳： 用于固定主控板、屏幕及传感器模块，使作品更加完整。

具体操作过程

  1. 传感器原型验证： 首先将土壤湿度传感器接入 ESP32-S2 Mini 的 ADC 引脚，通过串口读取空气中和水中的模拟量。 根据实际测试结果设置 dryValue = 4095 和 wetValue = 1500， 使湿度映射结果更加符合真实变化。
  2. TFT 屏幕调试： 屏幕使用软件 SPI 方式连接，分别设置 CS、DC、RST、SCLK、MOSI 引脚。调试过程中重点解决了 屏幕初始化、方向旋转、文字刷新闪烁等问题，最终采用局部矩形刷新方式减少屏幕闪烁。
  3. 蜂鸣器报警逻辑： 当湿度低于报警线且没有被手动静音时，程序利用 millis() 进行非阻塞式计时， 每 300ms 切换一次蜂鸣器状态，实现间歇报警，避免使用 delay() 影响网页响应。
  4. Web 控制台开发： 使用 WebServer server(80) 创建网页服务器，网页端每 3 秒自动刷新， 显示北京时间、实时湿度、报警状态、当前阈值和 AD 原始值，并提供 20%、30%、50% 三档报警线切换按钮。
  5. 网络与时间同步： 连接 WiFi 后通过阿里云 NTP 服务器同步北京时间，并在网页端和 TFT 屏幕上同时显示。 若 WiFi 连接失败，屏幕会提示离线模式，方便现场调试。
  6. 结构整理与最终演示： 完成面包板接线后，对杜邦线进行整理，并尝试通过 3D 打印结构件固定主控和屏幕。 最终演示时，使用滴水或将探头浸入水中的方式改变传感器读数，观察网页端、屏幕端和蜂鸣器的同步反馈。

项目实物与调试记录

技术难点与解决方案

  • 问题一：ADC 数值范围不稳定。 ESP32-S2 的 ADC 默认读取范围可能与校准值不一致，因此在 setup() 中加入 analogReadResolution(12)，统一为 0~4095 的 12 位读取范围。
  • 问题二：屏幕刷新闪烁。 如果每次循环都全屏刷新，会造成明显闪烁。最终采用 fillRect() 局部清除区域， 只重绘变化内容，提升显示稳定性。
  • 问题三：网页控制与蜂鸣器报警冲突。 若使用阻塞式延时，网页请求会卡顿。因此报警部分采用 millis() 计时， 在不影响 server.handleClient() 的情况下完成蜂鸣器闪烁报警。
  • 问题四：传感器干湿校准。 土壤湿度传感器在空气、水中和土壤中的 AD 值差别较大，因此通过实测空气干燥值和完全浸水值， 再使用 map() 和 constrain() 转换为百分比，保证网页显示直观易懂。

    html += "<a href='/set20'><button>设为缺水线: 20%</button></a><a href='/set30'><button>设为缺水线: 30%</button></a><a href='/set50'><button>设为缺水线: 50%</button></a></body></html>";
    return html;
    }

    // ================= Web服务器路由 =================
    void WebService() {
    server.on("/", []() { server.send(200, "text/html", HtmlPage()); });
    server.on("/mute", []() { isMuted = true; digitalWrite(BEEP_PIN, BEEP_OFF); server.send(200, "text/html", "<meta http-equiv='refresh' content='0; url=/'><p>已静音，正在返回...</p>"); });
    server.on("/set20", []() { alarmThreshold = 20; server.send(200, "text/html", "<meta http-equiv='refresh' content='0; url=/'><p>设置成功</p>"); });
    server.on("/set30", []() { alarmThreshold = 30; server.send(200, "text/html", "<meta http-equiv='refresh' content='0; url=/'><p>设置成功</p>"); });
    server.on("/set50", []() { alarmThreshold = 50; server.send(200, "text/html", "<meta http-equiv='refresh' content='0; url=/'><p>设置成功</p>"); });
    }

    void updateLocalTime() {
    struct tm timeinfo;
    if (!getLocalTime(&timeinfo)) return;
    sprintf(currentTimeStr, "%02d:%02d:%02d", timeinfo.tm_hour, timeinfo.tm_min, timeinfo.tm_sec);
    }

    void setup() {
    Serial.begin(115200);
    delay(100); 

    // ⚡️ S2 特有保护：强制将模拟量分辨率设为 12 位 (0-4095)，以匹配你之前的干湿校准值！
    analogReadResolution(12);
    
    pinMode(BEEP_PIN, OUTPUT);
    digitalWrite(BEEP_PIN, BEEP_OFF);

    // 屏幕初始化 (如果花屏依然可以换成 INITR_GREENTAB)
    tft.initR(INITR_BLACKTAB); 
    tft.setRotation(1);
    tft.fillScreen(ST77XX_BLACK);
    
    tft.setTextColor(ST77XX_GREEN);
    tft.setTextSize(2);
    tft.setCursor(15, 30);
    tft.print("Plant Guard");
    
    WiFi.begin(ssid, password);
    tft.setTextColor(ST77XX_WHITE);
    tft.setTextSize(1);
    tft.setCursor(15, 60);
    tft.print("Connecting WiFi...");
    
    int retry = 0;
    while (WiFi.status() != WL_CONNECTED && retry < 20) {
        delay(500);
        retry++;
    }

    tft.fillScreen(ST77XX_BLACK);
    if (WiFi.status() == WL_CONNECTED) {
        tft.setCursor(5, 20); tft.print("WiFi Connected!");
        tft.setCursor(5, 40); tft.print("IP: "); tft.print(WiFi.localIP());
        configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
        Serial.println(WiFi.localIP());
    } else {
        tft.setTextColor(ST77XX_RED);
        tft.setCursor(5, 20); tft.print("WiFi Failed!");
        tft.setCursor(5, 40); tft.print("Offline Mode");
    }
    delay(2000);
    tft.fillScreen(ST77XX_BLACK); 

    WebService();
    server.begin();
    }

    void loop() {
    server.handleClient();
    updateLocalTime();

    rawAnalogValue = analogRead(SOIL_PIN);
    currentMoisture = map(rawAnalogValue, dryValue, wetValue, 0, 100);
    currentMoisture = constrain(currentMoisture, 0, 100); 

    if (currentMoisture >= alarmThreshold) {
        isMuted = false;
    }

    bool shouldAlarm = (currentMoisture < alarmThreshold) && !isMuted;
    if (shouldAlarm) {
        if ((millis() / 300) % 2 == 0) digitalWrite(BEEP_PIN, BEEP_ON);
        else digitalWrite(BEEP_PIN, BEEP_OFF);
    } else {
        digitalWrite(BEEP_PIN, BEEP_OFF);
    }

    if (millis() - lastScreenUpdate > 500) {
        lastScreenUpdate = millis();

        // 屏幕防闪烁优化绘制
        tft.fillRect(0, 0, 160, 20, ST77XX_BLUE);
        tft.setTextColor(ST77XX_WHITE); 
        tft.setTextSize(1);
        tft.setCursor(5, 6);
        tft.print("Time: ");
        tft.print(currentTimeStr);

        tft.fillRect(0, 30, 160, 20, ST77XX_BLACK); 
        if (currentMoisture < alarmThreshold) {
        tft.setTextColor(ST77XX_RED);
        tft.setTextSize(2);
        tft.setCursor(15, 35);
        tft.print("NEED WATER!");
        } else {
        tft.setTextColor(ST77XX_GREEN);
        tft.setTextSize(1);
        tft.setCursor(20, 35);
        tft.print("Moisture Level:");
        }
        
        char moistureStr[10];
        sprintf(moistureStr, "%-3d%%", currentMoisture); 
        
        if (currentMoisture < alarmThreshold) {
            tft.setTextColor(ST77XX_RED, ST77XX_BLACK);
        } else {
            tft.setTextColor(ST77XX_GREEN, ST77XX_BLACK);
        }
        tft.setTextSize(4);
        tft.setCursor(40, 60);
        tft.print(moistureStr);

        tft.setTextSize(1);
        tft.setTextColor(ST77XX_WHITE, ST77XX_BLACK);
        tft.setCursor(5, 100);
        tft.print("IP: ");
        tft.print(WiFi.localIP().toString() + "    "); 
        
        tft.setCursor(5, 115);
        tft.print("WiFi Web Control Ready");
    }
    }
